    <div class="container">
        <div id="question-container">
            <p class="question" id="question-text">あなたが好きな風景はどちらですか？</p>
            <div class="image-options" id="options-container">
                <div class="image-option" onclick="selectOption(1)">
                    <img src="https://example.com/image1.jpg" alt="山の風景">
                </div>
                <div class="image-option" onclick="selectOption(0)">
                    <img src="https://example.com/image2.jpg" alt="海の風景">
                </div>
            </div>
        </div>
        <div id="result-container" style="display:none;">
            <p class="result" id="result-text"></p>
            <img id="result-image" class="result-image" src="" alt="結果の画像">
        </div>
    </div>
    <script>
        let score = 0;
        let currentQuestion = 0;
        const questions = [
            {
                text: "あなたが好きな風景はどちらですか？",
                images: ["https://example.com/image1.jpg", "https://example.com/image2.jpg"]
            },
            {
                text: "どちらの色が好きですか？",
                images: ["https://example.com/image3.jpg", "https://example.com/image4.jpg"]
            },
            {
                text: "どちらの活動が楽しそうですか？",
                images: ["https://example.com/image5.jpg", "https://example.com/image6.jpg"]
            }
        ];
        
        const results = [
            {
                text: "あなたは「冒険家タイプ」です！新しい体験を求める傾向があります。",
                image: "https://example.com/result1.jpg"
            },
            {
                text: "あなたは「リラックスタイプ」です。静かな環境で落ち着くことを好みます。",
                image: "https://example.com/result2.jpg"
            },
            {
                text: "あなたは「バランスタイプ」です。状況に応じて柔軟に対応できます。",
                image: "https://example.com/result3.jpg"
            }
        ];
        
        function updateQuestion() {
            document.getElementById("question-text").textContent = questions[currentQuestion].text;
            const optionsContainer = document.getElementById("options-container");
            optionsContainer.innerHTML = "";
            
            questions[currentQuestion].images.forEach((imgSrc, index) => {
                const optionDiv = document.createElement("div");
                optionDiv.className = "image-option";
                optionDiv.onclick = function() { selectOption(index); };
                
                const img = document.createElement("img");
                img.src = imgSrc;
                img.alt = `選択肢${index + 1}`;
                
                optionDiv.appendChild(img);
                optionsContainer.appendChild(optionDiv);
            });
        }
        
        function selectOption(value) {
            score += value;
            currentQuestion++;
            if (currentQuestion < questions.length) {
                updateQuestion();
            } else {
                showResult();
            }
        }
        
        function showResult() {
            document.getElementById("question-container").style.display = "none";
            document.getElementById("result-container").style.display = "block";
            
            let resultIndex = 0;
            if (score >= 2) {
                resultIndex = 0; // 冒険家タイプ
            } else if (score === 1) {
                resultIndex = 2; // バランスタイプ
            } else {
                resultIndex = 1; // リラックスタイプ
            }
            
            document.getElementById("result-text").textContent = results[resultIndex].text;
            document.getElementById("result-text").style.display = "block";
            document.getElementById("result-image").src = results[resultIndex].image;
        }
        
        // 初期表示
        updateQuestion();
    </script>
